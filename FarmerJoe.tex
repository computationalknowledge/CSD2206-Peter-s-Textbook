\section{DDL to generate the Tables}
\begin{verbatim}
CREATE TABLE lots (
lot_id varchar(20),
quantity_of_vegatables int,
veggie_id varchar(20),
lot_name varchar(20),
PRIMARY KEY (lot_id)
);

CREATE TABLE veggies(
veggie_id varchar(20),
veggie_price money,
veggie_name varchar(20),
veggie_type varchar(20)
PRIMARY KEY (veggie_id)
);
\end{verbatim}
